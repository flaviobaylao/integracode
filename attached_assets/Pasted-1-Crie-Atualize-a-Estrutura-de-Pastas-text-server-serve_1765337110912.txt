JavaScriptconst axios = require('axios');

async function perguntarGrok(mensagens, apiKey) {
  try {
    const response = await axios.post('https://api.x.ai/v1/chat/completions', {
      model: "grok-beta",
      messages: mensagens,
      temperature: 0.7,
      max_tokens: 800
    }, {
      headers: { Authorization: `Bearer ${apiKey}` }
    });
    return response.data.choices[0].message.content.trim();
  } catch (err) {
    console.error('Erro Grok:', err.response?.data || err.message);
    return "Desculpe, o assistente Grok está temporariamente indisponível.";
  }
}

module.exports = { perguntarGrok };

// Distribuição automática no webhook da Evolution
app.post('/webhook', (req, res) => {
  const mensagem = req.body;
  if (mensagem.event !== "messages.upsert") return res.sendStatus(200);
  const msg = mensagem.data;
  const numero = msg.key.remoteJid.split('@')[0];

  if (!conversas[numero]) {
    // Nova conversa → distribui automaticamente
    let maisLivre = null;
    let menorCarga = Infinity;
    Object.keys(atendentes).forEach(email => {
      if (atendentes[email].activeChats < menorCarga) {
        menorCarga = atendentes[email].activeChats;
        maisLivre = email;
      }
    });
    conversas[numero] = {
      assignedTo: maisLivre,
      unread: 1,
      nome: msg.pushName || "Cliente"
    };
    if (maisLivre) atendentes[maisLivre].activeChats++;
    io.emit('nova-conversa', { numero, assignedTo: maisLivre, nome: msg.pushName });
  } else if (!msg.key.fromMe) {
    conversas[numero].unread++;
    io.emit('nova-mensagem', { numero, msg });
  }

  res.sendStatus(200);
});

app.get('/api/buscar-leads', async (req, res) => {
  const { regiao, perfil, atividade } = req.query;
  const query = `${perfil} ${atividade || ''} em ${regiao}`;
  const apiKey = process.env.GOOGLE_PLACES_API_KEY;

  try {
    const response = await axios.get('https://maps.googleapis.com/maps/api/place/textsearch/json', {
      params: { query, key: apiKey, language: 'pt-BR' }
    });
    const leads = response.data.results.map(place => ({
      nome: place.name,
      endereco: place.formatted_address,
      telefone: place.formatted_phone_number || 'N/A',
      site: place.website || 'N/A',
      rating: place.rating || 'N/A'
    }));
    res.json({ leads });
  } catch (err) {
    res.status(500).json({ erro: 'Falha na busca' });
  }
});
